CONFIGURATION
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  OPENROUTER_API_KEY     = your_api_key (required)
  CODEX_MODEL            = model_name (optional)
  CODEX_TEMPERATURE      = 0-1 (optional)
  CODEX_MAX_ITERATIONS   = number (optional)
  CODEX_DEBUG            = true/false (optional)

TROUBLESHOOTING
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Problem: "codex: command not found"
Solution: pip install -e .

Problem: "OPENROUTER_API_KEY not set"
Solution: echo "OPENROUTER_API_KEY=your_key" > .env

Problem: ImportError
Solution: pip install -e . (reinstall)

For more help: See INSTALL.md
